Les différents types de films du réalisateur Martin Scorcesse
GENRE FILM REALISATEUR

SELECT g.type FROM genre g, film f WHERE g.id = f.idGenre GROUP BY g.type, g.id HAVING g.id = (SELECT f.idGenre FROM film f WHERE f.idRealisateur = (SELECT r.idRealisateur FROM realisateur r WHERE r.nomR = 'Scorcesse' AND r.prenomR = 'Martin'));
